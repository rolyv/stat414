      \item
	\begin{align*}
	 P(X > k+j | X > k) &= \dfrac{P(X > k+j \cap X > k)}{P(X > k)} \\
	    &= \dfrac{P(X > k+j)}{P(X > k)} \\
	    &= \dfrac{1 - P(X \le k + j)}{1 - P(X \le k)} \\
	    &= \dfrac{1 - (1 - p^{k+j})}{1 - (1 - p^k)} \\
	    &= \dfrac{p^{k+j}}{p^k} \\
	    &= p^j \\
	    &= 1 - (1 - p^j) \\
	    &= 1 - P(X \le j) \\
	    &= P(X > j)
	\end{align*}
